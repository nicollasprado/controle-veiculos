TABLE maintenance
COLUMN id INTEGER UNIQUE PRIMARY KEY GENERATED ALWAYS AS IDENTITY 
COLUMN vehicleName VARCHAR NOT NULL 
COLUMN dateTime TIMESTAMP NOT NULL 
COLUMN price DECIMAL NOT NULL 

TABLE users
COLUMN id UUID UNIQUE PRIMARY KEY DEFAULT uuid_generate_v4() 
COLUMN username VARCHAR NOT NULL 
COLUMN password VARCHAR NOT NULL 

TABLE vehicle
COLUMN id UUID UNIQUE PRIMARY KEY DEFAULT uuid_generate_v4() 
COLUMN name VARCHAR NOT NULL 

TABLE route
COLUMN id UUID UNIQUE PRIMARY KEY DEFAULT uuid_generate_v4() 
COLUMN startLocation VARCHAR NOT NULL 
COLUMN finalLocation VARCHAR NOT NULL 
COLUMN fuelRequiredPrice DECIMAL NOT NULL 
COLUMN otherCosts DECIMAL NOT NULL 

TABLE driver
COLUMN id UUID UNIQUE PRIMARY KEY DEFAULT uuid_generate_v4() 
COLUMN name VARCHAR NOT NULL 
COLUMN cpf VARCHAR NOT NULL
