TABLE maintenance
COLUMN id INTEGER UNIQUE PRIMARY KEY GENERATED ALWAYS AS IDENTITY 
COLUMN vehicle_name VARCHAR NOT NULL 
COLUMN date_time TIMESTAMP NOT NULL 
COLUMN price DECIMAL NOT NULL 

TABLE users
COLUMN id UUID UNIQUE PRIMARY KEY DEFAULT uuid_generate_v4() 
COLUMN username VARCHAR NOT NULL 
COLUMN password VARCHAR NOT NULL 

TABLE vehicle
COLUMN id UUID UNIQUE PRIMARY KEY DEFAULT uuid_generate_v4() 
COLUMN name VARCHAR NOT NULL 

TABLE route
COLUMN id INTEGER UNIQUE PRIMARY KEY GENERATED ALWAYS AS IDENTITY 
COLUMN start_location VARCHAR NOT NULL 
COLUMN final_location VARCHAR NOT NULL 
COLUMN fuel_required_price DECIMAL NOT NULL 
COLUMN other_costs DECIMAL NOT NULL 

TABLE driver
COLUMN id UUID UNIQUE PRIMARY KEY DEFAULT uuid_generate_v4() 
COLUMN name VARCHAR NOT NULL 
COLUMN cpf VARCHAR NOT NULL
